142  private void relinkNodes(Edge edgeToUpdate, List<IElement> refined)
143  {
144 
145  List<Node> newNodePath = new List<Node>();
146 
147  // assuming each element only has two allNodes on the path currently.
148 
149  // We want to add the new refined node that is closest to the other allNodes already in the path
150  // var distances = new List<double>();
151 
152  var theNewNodes = refined.SelectMany(x => x.getNodes());
153  HashSet<Node> refinedNodes = new HashSet<Node>(theNewNodes);
154 
155  Node[] pathNodes = edgeToUpdate.NodePath.ToArray();
156  Node currentNode;
157 
158  // loop through each of the gaps between two allNodes
159  for (int ii = 0; ii + 1 < pathNodes.Length; ii++)
160  {
161  var firstNode = pathNodes[ii];
162  var secondNode = pathNodes[ii + 1];
163 
164  currentNode = firstNode;
165 
166  // start with two very high initial values
167  List<double> distancesToSecond = new List<double>() {100000.0, 10000.0};
168 
169  // while the currentNode keeps getting closer to the second node and the search doesn't veer off
170  while(distancesToSecond[distancesToSecond.Count - 1] < distancesToSecond[distancesToSecond.Count - 2])
171  {
172 
173  // Get the distance between the current and all other allNodes
174  List<Tuple<Node, double>> comparisonsAgainstCurrent = refinedNodes
175  .Select(x => new Tuple<Node, double>(x, x.distanceTo(currentNode))).ToList();
176 
177 
178  List<Tuple<Node, double>> comparisonsAgainstSecond = refinedNodes
179  .Select(x => new Tuple<Node, double>(x, x.distanceTo(secondNode))).ToList();
180 
181 
182  // get the four closest to the first node, which of these is closes to the second
183  var fourClosestToCurrent = new List<Tuple<Node, double>>(comparisonsAgainstCurrent.OrderBy(x => x.Item2).Take(4));
184 
185 
186  List<Tuple<Node, double>> intersections = new List<Tuple<Node, double>>();
187 
188  foreach(var closeToCurr in fourClosestToCurrent)
189  {
190  foreach(var compAgainstSec in comparisonsAgainstSecond)
191  {
192  if(closeToCurr.Item1.Id == closeToCurr.Item1.Id)
193  {
194  intersections.Add(closeToCurr);
195  }
196  }
197  }
198 
199  var closestRoundCurrentToSecond = intersections.OrderBy(x => x.Item2).ToArray()[0];
200 
201  currentNode = closestRoundCurrentToSecond.Item1;
202 
203  // get rid of the node we have just added to the updated edge
204  refinedNodes.Remove(currentNode);
205  distancesToSecond.Add(currentNode.distanceTo(secondNode));
206  }
207  }
208 
209  newNodePath.Add(pathNodes[pathNodes.Length - 1]);
210 
211  edgeToUpdate.NodePath = newNodePath;
212  }

286  public double Percentile(double[] sequence, double excelPercentile)
287  {
288  Array.Sort(sequence);
289  int N = sequence.Length;
290  double n = (N - 1) * excelPercentile + 1;
291  // Another method: double n = (N + 1) * excelPercentile;
292  if (n == 1d) return sequence[0];
293  else if (n == N) return sequence[N - 1];
294  else
295  {
296  int k = (int)n;
297  double d = n - k;
298  return sequence[k - 1] + d * (sequence[k] - sequence[k - 1]);
